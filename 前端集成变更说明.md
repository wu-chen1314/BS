```typescript
import { createRouter, createWebHistory } from "vue-router";

        },
        {
          path: "favorites",

          meta: { title: "我的收藏" },
        },
        {
          path: "hot-ranking",
          name: "hot-ranking",
          component: () => import("../views/HotRanking.vue"),
          meta: { title: "热度排行" },
        },
        {
          path: "chat",
          name: "chat",
          component: () => import("../views/Chat.vue"),
          meta: { title: "AI 聊天" },
        },
        {

          name: "Person",
          component: () => import("../views/Profile.vue"),
          meta: { title: "个人中心" },

        // ===== 新增：地区分类路由 =====
        {
          path: "region-category",
          name: "region-category",
          component: () => import("../views/RegionCategory.vue"),
          meta: { title: "地区分类" },
        },
        // ===========================
      ],
    },
  ],
});

// 路由守卫...
router.beforeEach((to, from, next) => {
  const user = sessionStorage.getItem("user");
  const publicPages = ["/login"];
  const isPublicPage = publicPages.includes(to.path);

  if (!isPublicPage && !user) {
    next({ name: "login" });
  } else {
    next();
  }
});

export default router;
```

## 📊 API 接口依赖

页面依赖以下后端 API 接口：

| 接口 | 方法 | 说明 |
|------|------|------|
| `/api/region-category/all` | GET | 获取完整数据（页面初始化） |
| `/api/region-category/provinces` | GET | 获取所有省份 |
| `/api/region-category/cities` | GET | 获取城市列表 |
| `/api/region-category/categories/tree` | GET | 获取类别树 |
| `/api/region-category/projects` | GET | 获取项目列表 |
| `/api/region-category/statistics/by-region` | GET | 按地区统计 |
| `/api/region-category/statistics/by-category` | GET | 按类别统计 |

**后端服务地址**：`http://localhost:8080`（默认）

## 🔍 故障排查

### 问题 1：页面 404

**症状**：访问 `/region-category` 显示 404

**解决方案**：
1. 检查路由配置是否正确添加
2. 确认 `RegionCategory.vue` 文件已复制
3. 重启前端服务

### 问题 2：菜单不显示

**症状**：左侧导航菜单没有"地区分类"选项

**解决方案**：
1. 检查 `Layout.vue` 是否添加菜单项
2. 确认 `Location` 图标已导入
3. 检查菜单项 `index` 是否与路由 `path` 匹配

### 问题 3：数据加载失败

**症状**：页面显示"加载数据失败"

**解决方案**：
1. 确认后端服务已启动
2. 检查数据库是否已初始化
3. 查看浏览器控制台错误信息
4. 确认 API 地址配置正确

### 问题 4：导入错误

**症状**：`Failed to resolve import`

**解决方案**：
1. 运行 `npm install` 安装依赖
2. 检查 `@element-plus/icons-vue` 是否已安装
3. 重启开发服务器
